each 允许同时对多个元素进行循环，比如带分割线的区域 

            
                <template x-each="groups as g">
                    <a >
                        {{g.title}}  {{i}}
                    </a>
                    <b x-if="$index % 2 == 0">
                        {{g.desc}}
                    </b>
                    <br>
                </template>
