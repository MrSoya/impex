for 允许同时对多个元素进行循环，比如带分割线的区域 

            
                <template x-for="groups as g">
                    <a >
                        {{g.title}}  <span x-for="[1,2,3,4,5,6] as i">{{i}}</span>
                    </a>
                    <b x-if="$index % 2 == 0">
                        {{g.desc}}
                    </b>
                    <br>
                </template>
